.Params) 1 -}} {{- $path = .Get 0 -}} {{- if ge (len .Params) 2 -}} {{- $width = .Get 1 -}} {{- end -}} {{- end -}} {{- $postCardStyle := "" -}} {{- with $width -}} {{- $postCardStyle = printf "width: {{- $langPrefix := printf "/%s/" site.Language.Lang -}} {{- $candidates := slice $clean $noTrail (printf "%s/" $noTrail) -}} {{- if hasPrefix $clean $langPrefix -}} {{- $withoutLang := replaceRE $.Page.GetPage . -}} {{- if not $candidateTarget -}} {{- $candidateTarget = site.GetPage . -}} {{- end -}} {{- if $candidateTarget -}} {{- $scratch.Set "target" $candidateTarget -}} {{- end -}} {{- end -}} {{- end -}} {{- $target := $scratch.Get "target" -}} {{- if not $target -}} {{- errorf "[%s] %q: card shortcode target page not found for %q" site.Language.Lang .Page.Path $path -}} {{- end -}} {{- if eq $target.RelPermalink .Page.RelPermalink -}} {{- errorf "[%s] %q: card shortcode cannot target the current page (self-reference): %q" site.Language.Lang .Page.Path $path -}} {{- end -}}
